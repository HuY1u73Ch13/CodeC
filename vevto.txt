#include<bits/stdc++.h>
using namespace std;
long long n,a,i;
vector<int> b;
int main(){
     ios::sync_with_stdio(0);
    cin.tie(0); cout.tie(0);
    cin>>n;
    for (i = 1;i <= n;i++)
    {
        cin>>a;
        b.push_back(a);
    }
    for (i = 0; i < b.size();i++)
        if (b[i]%2 == 0) cout<<b[i]<<" ";
    cout<<'\n';
    for (i = b.size()-1;i >= 0;i--)
        if (b[i]%3 == 1) cout<<b[i]<<" ";
