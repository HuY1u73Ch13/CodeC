#include<bits/stdc++.h>
#define ll long long
using namespace std ;
ll n,s=1,a[1000005];
int main ()
{
    cin>>n;
    for(int i=1;i<=sqrt(n);i++)
    {
        if(n%i==0)
        {
            a[s]=i;
            s++;
        }
        if(i!=n/i and n%i==0)
        {
            a[s]=n/i;
            s++;
        }
    }
    s--;// bá» "0"
    sort(a+1,a+1+s);
    for(int i=1;i<=s;i++)
    {
        cout<<a[i]<<"\n";
    }
}
